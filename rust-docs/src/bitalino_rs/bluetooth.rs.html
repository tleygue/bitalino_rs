13const BTPROTO_RFCOMM: libc::c_int = 3;

15const DEFAULT_IO_TIMEOUT_SECS: u64 = 5;
16const MAX_CONNECT_RETRIES: u32 = 3;
17const RETRY_DELAY_MS: u64 = 500;

19/// High level connector that opens an RFCOMM socket without needing root.
20///
21/// The connector expects the device to already be paired/trusted (e.g., via
22/// `bluetoothctl`); you provide the MAC address and the optional PIN argument is
23/// ignored. Only a minimal libc-based stack is used.
24#[derive(Debug, Clone)]
25pub struct BluetoothConnector {
26    /// RFCOMM channel to connect to (BITalino default: 1).
27    pub channel: u8,
28    /// Per-operation I/O timeout applied to the RFCOMM socket.
29    pub io_timeout: Duration,
30    /// Maximum retry attempts for establishing RFCOMM.
31    pub max_retries: u32,
32    /// Delay between retries (exponential backoff uses this as the base).
33    pub retry_delay: Duration,
34}
35
36impl Default for BluetoothConnector {
37    fn default() -> Self {
38        Self {
39            channel: 1,
40            io_timeout: Duration::from_secs(DEFAULT_IO_TIMEOUT_SECS),
41            max_retries: MAX_CONNECT_RETRIES,
42            retry_delay: Duration::from_millis(RETRY_DELAY_MS),
43        }
44    }
45}
46
47impl BluetoothConnector {
48    /// Connect to an already-paired BITalino via RFCOMM using only libc sockets.
49    /// Caller must have paired and trusted the device ahead of time (e.g., via `bluetoothctl`).
50    pub fn pair_and_connect(&self, mac: &str, _pin: &str) -> Result<RfcommStream> {
51        let bdaddr = parse_bdaddr(mac)?;
52
53        let mut last_error = None;
54        for attempt in 0..self.max_retries {
55            if attempt > 0 {
56                let delay = self.retry_delay * (1 << (attempt - 1).min(3));
57                warn!(
58                    "retrying RFCOMM connection after {:?} (mac={}, attempt={})",
59                    delay, mac, attempt
60                );
61                thread::sleep(delay);
62            }
63
64            match open_rfcomm_raw(bdaddr, self.channel, self.io_timeout) {
65                Ok(stream) => {
66                    if let Err(e) = stream.verify_connected() {
67                        warn!("connection verification failed: mac={}, error={}", mac, e);
68                        last_error = Some(e);
69                        continue;
70                    }
71                    info!("RFCOMM connection established: mac={}", mac);
72                    return Ok(stream);
73                }
74                Err(e) => {
75                    warn!(
76                        "RFCOMM connection attempt failed: mac={}, attempt={}, error={}",
77                        mac, attempt, e
78                    );
79                    last_error = Some(e);
80                }
81            }
82        }
83
84        Err(last_error.unwrap_or_else(|| {
85            DriverError::Bluetooth(BluetoothError::NotConnected("max retries exceeded".into()))
86        }))
87    }
88}

90/// Simple RFCOMM stream that behaves like a Read/Write object.
91pub struct RfcommStream {
92    file: File,
93}
94
95impl RfcommStream {
96    /// Set the socket receive timeout (`SO_RCVTIMEO`).
97    ///
98    /// Used by the driver to enforce short user-supplied timeouts (e.g. in
99    /// `wait_until_streaming`) without waiting for the connector's default 5 s
100    /// kernel timeout to fire.
101    pub fn set_read_timeout(&self, timeout: Duration) -> std::io::Result<()> {
102        let tv = libc::timeval {
103            tv_sec: timeout.as_secs() as libc::time_t,
104            tv_usec: timeout.subsec_micros() as libc::suseconds_t,
105        };
106        let ret = unsafe {
107            libc::setsockopt(
108                self.file.as_raw_fd(),
109                libc::SOL_SOCKET,
110                libc::SO_RCVTIMEO,
111                &tv as *const _ as *const libc::c_void,
112                mem::size_of::<libc::timeval>() as libc::socklen_t,
113            )
114        };
115        if ret < 0 {
116            Err(std::io::Error::last_os_error())
117        } else {
118            Ok(())
119        }
120    }
121
122    /// Verify the connection is actually established and usable.
123    pub fn verify_connected(&self) -> Result<()> {
124        // Check socket error status
125        let mut err: libc::c_int = 0;
126        let mut len: libc::socklen_t = mem::size_of::<libc::c_int>() as libc::socklen_t;
127
128        let ret = unsafe {
129            libc::getsockopt(
130                self.file.as_raw_fd(),
131                libc::SOL_SOCKET,
132                libc::SO_ERROR,
133                &mut err as *mut _ as *mut libc::c_void,
134                &mut len,
135            )
136        };
137
138        if ret < 0 {
139            return Err(DriverError::Io(std::io::Error::last_os_error()));
140        }
141
142        if err != 0 {
143            return Err(DriverError::Bluetooth(BluetoothError::NotConnected(
144                std::io::Error::from_raw_os_error(err).to_string(),
145            )));
146        }
147
148        Ok(())
149    }
150}

173struct BdAddr {
174    b: [u8; 6],
175}
176
177#[repr(C)]
178struct SockAddrRc {
179    rc_family: libc::sa_family_t,
180    rc_bdaddr: BdAddr,
181    rc_channel: u8,
182}
183
184fn parse_bdaddr(mac: &str) -> Result<BdAddr> {
185    let parts: Vec<&str> = mac.split(':').collect();
186    if parts.len() != 6 {
187        return Err(DriverError::Bluetooth(BluetoothError::Connection(
188            "invalid mac".into(),
189        )));
190    }
191
192    let mut bytes = [0u8; 6];
193    for (i, part) in parts.iter().enumerate() {
194        let byte = u8::from_str_radix(part, 16).map_err(|_| {
195            DriverError::Bluetooth(BluetoothError::Connection("invalid mac".into()))
196        })?;
197        bytes[i] = byte;
198    }
199
200    // bdaddr_t stores bytes in reverse order compared to the usual MAC string
201    let mut addr = BdAddr { b: [0; 6] };
202    for i in 0..6 {
203        addr.b[i] = bytes[5 - i];
204    }
205    Ok(addr)
206}

208fn open_rfcomm_raw(address: BdAddr, channel: u8, timeout: Duration) -> Result<RfcommStream> {
209    debug!(
210        "opening RFCOMM socket: channel={}, addr_bytes={:02X?}",
211        channel, address.b
212    );
213
214    let fd = unsafe {
215        libc::socket(
216            AF_BLUETOOTH as libc::c_int,
217            libc::SOCK_STREAM,
218            BTPROTO_RFCOMM,
219        )
220    };
221    if fd < 0 {
222        return Err(DriverError::Bluetooth(BluetoothError::Connection(
223            std::io::Error::last_os_error().to_string(),
224        )));
225    }
226
227    if unsafe { libc::fcntl(fd, libc::F_SETFD, libc::FD_CLOEXEC) } < 0 {
228        let err = std::io::Error::last_os_error();
229        unsafe {
230            libc::close(fd);
231        }
232        return Err(DriverError::Bluetooth(BluetoothError::Connection(
233            err.to_string(),
234        )));
235    }
236
237    let mut addr = SockAddrRc {
238        rc_family: AF_BLUETOOTH as libc::sa_family_t,
239        rc_bdaddr: address,
240        rc_channel: channel,
241    };
242
243    let ret = unsafe {
244        libc::connect(
245            fd,
246            &mut addr as *mut _ as *const libc::sockaddr,
247            mem::size_of::<SockAddrRc>() as libc::socklen_t,
248        )
249    };
250    if ret < 0 {
251        let err = std::io::Error::last_os_error();
252        unsafe {
253            libc::close(fd);
254        }
255        return Err(DriverError::Bluetooth(BluetoothError::Connection(
256            err.to_string(),
257        )));
258    }
259
260    // Set IO timeouts to avoid hanging reads/writes.
261    let tv = libc::timeval {
262        tv_sec: timeout.as_secs() as libc::time_t,
263        tv_usec: timeout.subsec_micros() as libc::suseconds_t,
264    };
265    for opt in [libc::SO_RCVTIMEO, libc::SO_SNDTIMEO] {
266        let ret = unsafe {
267            libc::setsockopt(
268                fd,
269                libc::SOL_SOCKET,
270                opt,
271                &tv as *const _ as *const libc::c_void,
272                mem::size_of::<libc::timeval>() as libc::socklen_t,
273            )
274        };
275        if ret < 0 {
276            let err = std::io::Error::last_os_error();
277            unsafe {
278                libc::close(fd);
279            }
280            return Err(DriverError::Bluetooth(BluetoothError::Connection(
281                err.to_string(),
282            )));
283        }
284    }
285
286    let file = unsafe { File::from_raw_fd(fd) };
287    Ok(RfcommStream { file })
